#include<iostream>

    }

### 滑雪

![](2025-12-16-21-40-23.png)

```cpp
#include<iostream>
#include<algorithm>
using namespace std;
const int N=110;
int n,m,st[N][N],h[N][N],res=0;
int d1[4]={-1,0,1,0},d2[4]={0,1,0,-1};

int dfs(int x,int y){
    if(st[x][y])return st[x][y];
    st[x][y]=1;
    for(int i=0;i<4;i++){
        int nx=x+d1[i];
        int ny=y+d2[i];
        if(nx>=0 && nx<n && ny>=0 && ny<m && h[nx][ny]<h[x][y])st[x][y]=max(st[x][y],dfs(nx,ny)+1);
    }
    return st[x][y];
}
int main(){
    scanf("%d%d",&n,&m);
    for(int i=0;i<n;i++){
        for(int j=0;j<m;j++){
        scanf("%d",&h[i][j]);
        }
    }
    for(int i=0;i<n;i++){
        for(int j=0;j<m;j++){
        res=max(dfs(i,j),res);
        }
    }    
    printf("%d",res);
    return 0;
}
```


### 词语接龙

![](2025-12-16-22-51-43.png)

```cpp

#include<iostream>
#include<algorithm>
using namespace std;
const int N=50;
int cnt[N],n,res=0;
string s[N];
void dfs(const string& tmp){
    res=max(res,(int)tmp.size());
    for(int i=0;i<n;i++){
        if(cnt[i]>=2)continue;
        for(int j=1;j<min((int)tmp.size(),(int)s[i].size());j++){
            if(tmp.substr(tmp.size()-j)==s[i].substr(0,j)){
                cnt[i]++;
                dfs(tmp+s[i].substr(j));
                cnt[i]--;
            }
        }
    }
}


int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    cin>>n;
    for(int i=0;i<n;i++)cin>>s[i];
    char c;
    cin>>c;
    for(int i=0;i<n;i++){
        if(s[i][0]==c){
            cnt[i]++;
            dfs(s[i]);
            cnt[i]--;
        }
    }
    cout<<res<<endl;
    return 0;
}
```

## DDL

### 查询所有数据库
show databases;

### 查询当前数据库
select database();

### 创建数据库
create database if not exists db1;
可以自定义字符集
create database if not exists db1 charset utf8mb4;

### 删除数据库
drop database if exists db1;

### 切换数据库
use db1;

### 查询当前数据库所有表
show tables;

### 创建表
create table if not exists t1(
    id int comment '主键',
    name varchar(20) comment '姓名',
    age int comment '年龄'
);

desc t1;

drop table if exists t1;
