

### Floyd算法

类似dp，但是是多阶段决策
```cpp
for (k = 1; k <= n; k++) {
  for (x = 1; x <= n; x++) {
    for (y = 1; y <= n; y++) {
      f[k][x][y] = min(f[k - 1][x][y], f[k - 1][x][k] + f[k - 1][k][y]);
    }
  }
}
```

初始化dis数组，将original -> changed 映射为边权，得到多源最短路径，最后一一替换原字符串即可

```cpp


class Solution {
public:
    long long minimumCost(string source, string target, vector<char>& original, vector<char>& changed, vector<int>& cost) {
        int dis[26][26];
        const int INF = 0x3f3f3f3f;
        memset(dis,0x3f,sizeof dis);
        int n = original.size();
        for(int i = 0; i < 26; i++)dis[i][i] = 0;
        for(int i = 0; i < n; i++){
            int u = original[i] - 'a';
            int v = changed[i] - 'a';
            dis[u][v] = min(dis[u][v],cost[i]);
        }
        for(int k = 0; k < 26; k++){
            for(int i = 0; i < 26; i++){
                if(dis[i][k] == INF)continue;

                for(int j = 0; j < 26; j++){
                    dis[i][j] = min(dis[i][j],dis[i][k] + dis[k][j]);
                }
            }
        }
        long long ans = 0;
        int m = source.size();
        for(int i = 0; i < m; i++){
            int u = source[i] - 'a';
            int v = target[i] - 'a';
            if(dis[u][v] == INF)return -1;
            ans +=dis[u][v];
        }
        return ans;
    }
```
